# charset(str) 
返回字符串所用的字符集
```
SELECT CHARSET(ename) FROM emp;
```
# CONCAT(string2 \[,....]) 
连接字符串,将多个列拼接为一个列
```
SELECT CONCAT (ename,' job is ' ,job) FROM emp;
```
# INSERT(string,substring)
返回substring在string 中出现的位置,没有返回0
```
SELECT INSERT('zhangyong0325','yong') FROM DUAL;
```
dual 亚元表 系统表,可作为测试表使用
# UCASE(string) 
转换乘大写
```
SELECT UCASE(ename) FROM emp;
```
# LCASE(string) 
转换为小写
```
SELECT LCASE(ename) FROM emp;
```
# LEFT(string2,length) 
从string2中的左边起取length个字符
```
SELECT LEFT(ename,2) FROM emp;
```
# RIGHT(string2,length) 
从string2中的右边起取length个字
```
SELECT RIGHT(ename,2) FROM emp;
```

# LENGTH(string) 
计算string长度\[按照字节]
```
SELECT LENGTH(ename) FROM emp;
SELECT LENGTH('张勇') FROM emp;
```
# REPLACE(str,search_str,replace_str) 
在str中用replace_str替换search_str
```
SELECT ename,REPLACE(job,'MANAGER','经理') FROM emp;
```
# STRCMP(str1,str2) 
逐字符比较两个字符串大小
==是否区分大小写与校验规则有关==
```
SELECT STRCMP('zhangyong','Zhangyong') FROM DUAL;
SELECT STRCMP('zhangyong','zhangzhang') FROM DUAL;
```
# SUBSTRING(str,position,\[, length]) 
在str的position的位置向左取length个字符
```
SELECT SUBSTRING(ename,1,2) FROM emp;
```
# LTRIM(string) RTRIM(string) TRIM(string)
去除左空格,去除右空格,去除两端空格
```
SELECT LTRIM('   yonghe   ') FROM DUAL;
SELECT RTRIM('   yonghe   ') FROM DUAL;
SELECT TRIM('   yonghe   ') FROM DUAL;
```
